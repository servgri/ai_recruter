SELECT DEPARTMENT_ID,
 DEPARTMENT_NAME,
 SUM(SALARY) TOTAL_SALARY
FROM EMPLOYEE E, 
 DEPARTMENTS D
WHERE E.DEPARTMENT_ID = D.DEPARTMENT_ID 
 AND MANAGER_ID = NULL
 AND LOCATION_ID = (SELECT LOCATION_ID
 FROM LOCATIONS 
 WHERE CITY = 'SEOUL')
 AND SUM(SALARY) >= 100000
GROUP BY DEPARTMENT_NAME
