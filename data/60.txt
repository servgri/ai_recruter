CREATE TABLE clients (
    client_id      SERIAL PRIMARY KEY,
    client_name    VARCHAR(255) NOT NULL,
    client_surname VARCHAR(255) NOT NULL,
    login          VARCHAR(30)  NOT NULL UNIQUE,
    city_id        INT,
    birth_date     DATE,
    reg_date       DATE NOT NULL DEFAULT CURRENT_DATE,
    age INT GENERATED ALWAYS AS (EXTRACT(YEAR FROM AGE(reg_date, birth_date))) STORED,
    FOREIGN KEY (city_id) REFERENCES cities(city_id),
    CONSTRAINT valid_birth_date CHECK (birth_date <= CURRENT_DATE)
);

SELECT 
    d.department_id,
    d.department_name,
    SUM(e.salary) AS total_salary
FROM employees AS e
JOIN departments AS d ON e.department_id = d.department_id
WHERE e.manager_id IS NULL
  AND d.location_id = (SELECT location_id FROM locations WHERE city = 'Seoul')
GROUP BY d.department_id, d.department_name
HAVING SUM(e.salary) > 100000;
