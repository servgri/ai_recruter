3. Запрос должен иметь следующий вид:
	SELECT
		DEPARTMENT_ID,
		DEPARTMENT_NAME,
		SUM(E.SALARY) AS TOTAL_SALARY
	FROM EMPLOYEE E
	INNER JOIN DEPARTMENT D
		ON E.DEPARTMENT_ID = D.DEPARTMENT_ID --соединение таблиц через join
	WHERE 
		E.MANAGER_ID IS NULL --правильное сравнение null 
		AND D.LOCATION_ID = (
			SELECT LOCATION_ID
			FROM LOCATIONS
			WHERE CITY = 'SEOUL'
		)
	GROUP BY
		DEPARTMENT_ID, DEPARTMENT_NAME
	HAVING SUM(SALARY) > 100000
