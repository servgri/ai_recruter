SELECT 
   D.DEPARTMENT_ID,
   D.DEPARTMENT_NAME,
   SUM(E.SALARY) AS TOTAL_SALARY
FROM EMPLOYEE E
  JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
    JOIN LOCATIONS L ON D.LOCATION_ID = L.LOCATION_ID
WHERE 
      E.MANAGER_ID IS NULL,
  AND L.CITY = 'SEOUL'
GROUP BY
    D.DEPARTMENT_ID, 
    D.DEPARTMENT_NAME
HAVING
    SUM(E.SALARY) > 100000;
