clients(

Необходимо создать таблицу для хранения истории цен, таблицу причин изменения цен(акция, сезонное повышение цен и т.д.)

items(
item_id number primary key,
item_name varchar(255) not null
);

prices(
price_id number primary key,
item_id not null references items(item_id),
cost number not null check(cost>0),
installation_date date not null,
cancellation_date date check (cancellation_date is null or cancellation_date > installation_date),
reason_id number refernces change_price_reasons(reason_id)
);

change_price_reasons(
reason_id number primary key,
name varchar(255) not null,
description varchar(255),
)

Задание 3.
Исправленный запрос:
select d.department_id, d.department_name, sum(e.salary) as total from departments d join employees e on d.department_id=e.department_id join locations l on d.location_id=l.location_id 
where l.city="SEOUL" and e.manager_id is null 
group by d.department_id, d.department_name
having sum(salary)>100000
