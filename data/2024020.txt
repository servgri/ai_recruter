3.      SELECT D.DEPARTMENT_ID,
 		D.DEPARTMENT_NAME,
 		SUM(E.SALARY) TOTAL_SALARY
	FROM EMPLOYEE E
        	JOIN DEPARTMENTS D ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
		JOIN LOCATIONS L ON L.LOCATION_ID = D.LOCATION_ID
	WHERE 
		E.MANAGER_ID IS NULL
 		AND L.CITY = 'SEOUL'
	GROUP BY
		D.DEPARTMENT_ID,
		D.DEPARTMENT_NAME
	HAVING SUM(E.SALARY) >= 100000
