SELECT DEPARTMENT_ID,
 DEPARTMENT_NAME
FROM EMPLOYEE E,
 DEPARTMENTS D
WHERE E.DEPARTMENT_ID = D.DEPARTMENT_ID
 AND MANAGER_ID = NULL
 AND LOCATION_ID = (SELECT LOCATION_ID
 FROM LOCATIONS
 WHERE CITY = 'SEOUL')
GROUP BY DEPARTMENT_ID,DEPARTMENT_NAME
HAVING SUM(SALARY) >= 100000
